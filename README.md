# BDM-interface
 Implementazione di un'interfaccia BDM con Raspberry Pi Pico
 
# Lista comandi 
- 'q':  ACK_ENABLE
- 'w':  ACK_DISABLED
- 'e':  BACKGROUND
- 'r':  GO
- 't':  TRACE1
- 'y':  TAGGO
---
- 'a':  READ_A
- 's':  READ_CCR
- 'd':  READ_PC
- 'f':  READ_HX
- 'g':  READ_SP
- 'h':  READ_NEXT
- 'j':  READ_NEXT_WS
---
- 'z':  WRITE_A
- 'x':  WRITE_CCR
- 'c':  WRITE_PC
- 'v':  WRITE_HX
- 'b':  WRITE_SP
- 'n':  WRITE_NEXT
- 'm':  WRITE_NEXT_WS
---
- '1':  READ_STATUS
- '2':  WRITE_CONTROL
- '3':  READ_BYTE
- '4':  READ_BYTE_WS
- '5':  READ_LAST
- '6':  WRITE_BYTE
- '7':  WRITE_BYTE_WS
- '8':  READ_BKPT
- '9':  WRITE_BKPT
